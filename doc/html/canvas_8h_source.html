76 void createLine(Vector2f vec, RGBAColor color) {
77 float x0 = 0.0f;
78 float y0 = 0.0f;
79 float x1 = vec.x;
80 float y1 = vec.y;
81
82 bool steep = false;
83 if (std::abs(x0-x1) < std::abs(y0-y1)) {
84 std::swap(x0, y0);
85 std::swap(x1, y1);
86 steep = true;
87 }
88 if (x0 > x1) {
89 std::swap(x0, x1);
90 std::swap(y0, y1);
91 }
92 int dx = x1-x0;
93 int dy = y1-y0;
94 int derror2 = std::abs(dy)*2;
95 int error2 = 0;
96 int y = y0;
97 for (int x = x0; x <= x1; x++) {
98 if (steep) {
99 this->pixels.push_back( Pixel(Point2i(y,x), color) );
100 } else {
101 this->pixels.push_back( Pixel(Point2i(x,y), color) );
102 }
103 error2 += derror2;
104
105 if (error2 > dx) {
106 y += (y1 > y0 ? 1 : -1);
107 error2 -= dx*2;
108 }
109 }
